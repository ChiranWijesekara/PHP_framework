CREATE TABLE Users(
    id INT AUTO_INCREMENT,
    name VARCHAR(255),
    email VARCHAR(255),
    password VARCHAR(255),
    create_at DATETIME CURREnt_TIMESTAMP,
    PRIMARY KEY(id);

);

CREATE TABLE Posts(
    id INT AUTO_INCREMENT,
    user_id INT,
    title VARCHAR(255),
    body TEXT,
    PRIMARY KEY(id),
    FOREIGN KEY (user_id) REFERENCES Users(id)  ON DELETE CASCADE
);
